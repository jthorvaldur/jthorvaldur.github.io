PostgreSQL agreed core

DDL for review

Only agreed directions are here. Rollout must still be staged: add nullable links, backfill/group historical rows, validate balances, then enforce requirements.

-- Agreed core candidate v0.2 for Volk review; not approved for production.
CREATE EXTENSION IF NOT EXISTS pgcrypto;

CREATE TABLE ledger_events (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  event_key text NOT NULL UNIQUE,
  event_type text NOT NULL,
  status text NOT NULL CHECK (status IN ('pending','posted','reversed','failed')),
  order_id integer REFERENCES orders(id),
  source_type text NOT NULL,
  source_id text NOT NULL,
  effective_at timestamptz NOT NULL,
  posted_at timestamptz NOT NULL DEFAULT now(),
  corrects_event_id uuid REFERENCES ledger_events(id),
  metadata jsonb NOT NULL DEFAULT '{}'::jsonb
);

ALTER TABLE ledger_entries
  ADD COLUMN ledger_event_id uuid REFERENCES ledger_events(id),
  ADD COLUMN line_number integer,
  ADD COLUMN account_code text,
  ADD COLUMN effective_at timestamptz,
  ADD COLUMN posted_at timestamptz NOT NULL DEFAULT now();

CREATE UNIQUE INDEX uq_ledger_event_line
  ON ledger_entries (ledger_event_id, line_number)
  WHERE ledger_event_id IS NOT NULL;
CREATE INDEX idx_ledger_effective_at ON ledger_entries (effective_at);
CREATE INDEX idx_ledger_posted_at ON ledger_entries (posted_at);

CREATE TABLE stripe_balance_transactions (
  stripe_id text PRIMARY KEY,
  source_object_id text,
  source_type text NOT NULL,
  currency text NOT NULL,
  gross_minor bigint NOT NULL,
  fee_minor bigint NOT NULL,
  net_minor bigint NOT NULL,
  status text NOT NULL,
  available_on timestamptz,
  observed_at timestamptz NOT NULL DEFAULT now(),
  raw_snapshot jsonb NOT NULL,
  CHECK (gross_minor - fee_minor = net_minor)
);

Open design appendix

Shapes to discuss, not migrate

These sketches preserve the previous technical thinking without claiming that separate responsibility/remedy tables—or one versus six posting tables—have been chosen.

-- OPTIONAL SHAPES ONLY: no decision has been made to add these tables.
CREATE TABLE responsibility_decisions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  order_item_id integer NOT NULL REFERENCES order_items(id),
  actor_type text NOT NULL,
  actor_id text NOT NULL,
  quantity numeric(14,3) NOT NULL CHECK (quantity >= 0),
  reason text NOT NULL,
  status text NOT NULL,
  decision_version integer NOT NULL CHECK (decision_version > 0),
  supersedes_id uuid REFERENCES responsibility_decisions(id),
  effective_at timestamptz NOT NULL,
  posted_at timestamptz NOT NULL DEFAULT now(),
  evidence jsonb NOT NULL DEFAULT '[]'::jsonb,
  UNIQUE (order_item_id, actor_type, actor_id, decision_version)
);

CREATE TABLE remedy_lines (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  resolution_id integer NOT NULL REFERENCES pickup_resolutions(id),
  order_item_id integer NOT NULL REFERENCES order_items(id),
  quantity numeric(14,3) NOT NULL CHECK (quantity > 0),
  merchandise_minor bigint NOT NULL CHECK (merchandise_minor >= 0),
  tax_minor bigint NOT NULL CHECK (tax_minor >= 0),
  fixed_fee_minor bigint NOT NULL CHECK (fixed_fee_minor >= 0),
  method text NOT NULL CHECK (method IN ('original_tender','credit','replacement')),
  tax_components jsonb NOT NULL DEFAULT '[]'::jsonb,
  effective_at timestamptz NOT NULL,
  posted_at timestamptz NOT NULL DEFAULT now()
);
